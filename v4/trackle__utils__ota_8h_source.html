210 int firmware_ota_url(const char *url, uint32_t crc)
211 {
212  // check if reset start_timestamp

214  if (bits & OTA_UPDATING)

220  if (xOtaTaskHandle != NULL)

222  vTaskDelete(xOtaTaskHandle);
223  }
224  xEventGroupClearBits(s_wifi_event_group, OTA_UPDATING);
225  }
226  else // return error

228  return OTA_ERR_ALREADY_RUNNING;
229  }
