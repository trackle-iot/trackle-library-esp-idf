108 /* Event handler for catching system events */
109 static void bt_event_handler(void *arg, esp_event_base_t event_base,
110  int32_t event_id, void *event_data)
111 {
112  ESP_LOGI(BT_TAG, "----------------------------------------");
113  ESP_LOGI(BT_TAG, "bt event_handler: %s %" PRIu32, event_base, event_id);
114  ESP_LOGI(BT_TAG, "----------------------------------------");
115 
116 #ifdef PROTOCOMM_EVENTS_SUPPORTED
117  if (event_base == PROTOCOMM_TRANSPORT_BLE_EVENT)
118  {
119  switch (event_id)
120  {
121  case PROTOCOMM_TRANSPORT_BLE_CONNECTED:
122  ESP_LOGI(BT_TAG, "PROTOCOMM SESSION STARTED");
123  xEventGroupSetBits(wifiProvisioningEvents, PROV_PROTOCOMM_SESSION_READY);
124  break;
125  case PROTOCOMM_TRANSPORT_BLE_DISCONNECTED:
126  ESP_LOGI(BT_TAG, "PROTOCOMM SESSION STOPPED");
127  xEventGroupClearBits(wifiProvisioningEvents, PROV_PROTOCOMM_SESSION_READY);
128  break;
129  }
130  }
131 #endif
132  if (event_base == WIFI_PROV_EVENT)
133  {
134  switch (event_id)
135  {
136  case WIFI_PROV_START:
137  ESP_LOGI(BT_TAG, "Provisioning started");
138  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN | PROV_EVT_CRED);
139  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_RUN);
140  break;
141  case WIFI_PROV_CRED_RECV:
142  {
143  wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
144  ESP_LOGI(BT_TAG, "Received Wi-Fi credentials"
145  "\n\tSSID : %s\n\tPassword : %s",
146  (const char *)wifi_sta_cfg->ssid,
147  (const char *)wifi_sta_cfg->password);
148  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_CRED);
149  break;
150  }
151  case WIFI_PROV_CRED_FAIL:
152  {
153  wifi_prov_sta_fail_reason_t *reason = (wifi_prov_sta_fail_reason_t *)event_data;
154  ESP_LOGE(BT_TAG, "Provisioning failed!\n\tReason : %s"
155  "\n\tPlease reset to factory and retry provisioning",
156  (*reason == WIFI_PROV_STA_AUTH_ERROR) ? "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
157 
158  prov_retry_num++;
159  if (prov_retry_num >= PROV_MGR_MAX_RETRY_CNT)
160  {
161  ESP_LOGI(BT_TAG, "Failed to connect with provisioned AP, reseting provisioned credentials and restarting...");
162  wifi_config_t wifi_cfg = {0};
163  esp_err_t err = esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
164  if (err != ESP_OK)
165  {
166  ESP_LOGE(BT_TAG, "Failed to set wifi config, 0x%x", err);
167  }
168  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN | PROV_EVT_CRED);
169  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_ERR);
170  xEventGroupSetBits(s_wifi_event_group, RESTART);
171  }
172 
173  break;
174  }
175  case WIFI_PROV_CRED_SUCCESS:
176  ESP_LOGI(BT_TAG, "Provisioning successful");
177  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN | PROV_EVT_CRED);
178  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_OK);
179  break;
180  case WIFI_PROV_END:
181  // De-initialize manager once provisioning is finished and restart
182  ESP_LOGI(BT_TAG, "Provisioning end");
183  wifi_prov_mgr_deinit();
184  xEventGroupClearBits(wifiProvisioningEvents, PROV_EVT_NO | PROV_EVT_OK | PROV_EVT_ERR | PROV_EVT_RUN | PROV_EVT_CRED);
185  xEventGroupSetBits(wifiProvisioningEvents, PROV_EVT_NO);
186  xEventGroupSetBits(s_wifi_event_group, RESTART);
187  break;
188  default:
189  break;
190  }
191  }
192 
193  ESP_LOGI(BT_TAG, "end bt_event_handler: -------------------");
194 }

212 static int btPostCbClaimCode(const char *args)
213 {
214  char *key = strtok(args, ",");
215  if (key == NULL || strcmp(key, "cc") != 0)
216  {
217  ESP_LOGE("cc", "Invalid key for setting claim code");
218  return -1;
219  }
220  char *claimCode = strtok(NULL, ",");
221  if (key == NULL || strlen(claimCode) != 63)
222  {
223  ESP_LOGE("cc", "Invalid claim code");
224  return -1;
225  }
226  ESP_LOGE("cc", "Claim code received successfully:");
227  ESP_LOG_BUFFER_CHAR_LEVEL("cc", claimCode, CLAIM_CODE_LENGTH, ESP_LOG_ERROR);
228  trackleSetClaimCode(trackle_s, claimCode);
229  Trackle_saveClaimCode(claimCode);
230  return 1;
231 }

271 void trackle_utils_bt_provision_loop()
272 {
273  EventBits_t bits = xEventGroupGetBits(s_wifi_event_group);
274 
275  // Check if start or stop provisioning
276  if (bits & START_PROVISIONING)
277  {
278  xEventGroupClearBits(s_wifi_event_group, START_PROVISIONING);
279  xEventGroupSetBits(s_wifi_event_group, IS_PROVISIONING);
280 
281  esp_wifi_set_ps(WIFI_PS_MIN_MODEM); // enable powersave
282 
283  // Configuration for the provisioning manager
284  wifi_prov_mgr_config_t config = {
285  .scheme = wifi_prov_scheme_ble,
286  .scheme_event_handler = WIFI_PROV_SCHEME_BLE_EVENT_HANDLER_FREE_BTDM,
287  };
288 
289  // Initialize provisioning manager
290  wifi_prov_mgr_init(config);
291 
292  btFunctionsEndpointsCreate();
293 
294  wifi_prov_scheme_ble_set_service_uuid(bleProvUuid);
295 
296  if (bleAdvDataLen > 0)
297  {
298  const esp_err_t e = wifi_prov_scheme_ble_set_mfg_data(bleAdvData, bleAdvDataLen);
299  ESP_LOGE("", "ERROR REG ADV: %s", esp_err_to_name(e));
300  }
301 
302  if (strlen(bleProvDeviceName) == 0)
303  {
304  uint8_t eth_mac[6];
305  char default_name[20];
306  esp_wifi_get_mac(WIFI_IF_STA, eth_mac);
307  snprintf(default_name, 20, "DEVICE_%02X%02X%02X", eth_mac[3], eth_mac[4], eth_mac[5]);
308  trackle_utils_bt_provision_set_device_name(default_name);
309  }
310 
311  esp_err_t prov_err = wifi_prov_mgr_start_provisioning(WIFI_PROV_SECURITY_1, NULL, bleProvDeviceName, NULL);
312  ESP_LOGI(BT_TAG, "wifi_prov_mgr_start_provisioning %d", prov_err);
313  btFunctionsEndpointsRegister();
314  }
315 }
